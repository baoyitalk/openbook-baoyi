function mergeStyles(...styles) {
  return styles.reduce((merged, style) => {
    if (!style) return merged;
    if (typeof style === 'object') {
      return { ...merged, ...style };
    }
    return merged;
  }, {});
}

function Component({ isActive, isLarge, customStyle }) {
  const baseStyle = { padding: '10px', borderRadius: '4px' };
  const activeStyle = isActive && { background: 'blue', color: 'white' };
  const largeStyle = isLarge && { fontSize: '18px', padding: '15px' };

  const style = mergeStyles(baseStyle, activeStyle, largeStyle, customStyle);

  return <div style={style}>Content</div>;
}
